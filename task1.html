Task 1 - Number Palindrome

using System;
public class P
{
    public static void Main (string [] args)
    {
        int n, r, sum = 0, temp;
        Console.Write("Enter the Number: ");
        n = int.Parse(Console.ReadLine());
        temp = n;
        while(n > 0)
        {
            r = n % 10;
            sum = (sum * 10) + r;
            n = n / 10;
        }
        if(temp == sum)
            Console.Write("Number is Palindrome.");
        else
            Console.Write("Number is not Palindrome");
    }
}
Steps to Compile & Run:
  1. Save the code in a file named P.cs
  2. Open terminal and type: csc P.cs
  3. Then run using: .\P
