Task 6 - Strings

This program demonstrates the difference between String and StringBuilder when performing concatenation operations in C#.

üßæ Code

using System;
using System.Text;

class stringbuilder
{
    // Concatenates to String
    public static void concat1(String s1)
    {
        // Concatenation using String.Concat
        String st = "To C#.NET Lab";
        s1 = String.Concat(s1, st);
        // Alternatively: s1 = s1 + "To C#.NET Lab";
    }

    // Concatenates to StringBuilder
    public static void concat2(StringBuilder s2)
    {
        s2.Append("To C#.NET Lab");
    }

    public static void Main(String[] args)
    {
        String s1 = "Welcome";
        concat1(s1); // s1 is not changed
        Console.WriteLine("Using String Class: " + s1);

        StringBuilder s2 = new StringBuilder("Welcome");
        concat2(s2); // s2 is changed
        Console.WriteLine("Using StringBuilder Class: " + s2);
    }
}
Note: In C#, strings are immutable, so concatenation with a normal string does not change the original string unless reassigned. However, StringBuilder is mutable and supports efficient modifications.
