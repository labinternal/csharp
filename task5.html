Task 5 - Window Form using HTML Controls

Step-by-step guide:
  1. Open terminal and create a new Razor web app:
    dotnet new webapp -n InvoiceApp

üìÑ index.cshtml.cs

using Microsoft.AspNetCore.Mvc.RazorPages;
using System.Collections.Generic;

namespace InvoiceApp.Pages
{
    public class IndexModel : PageModel
    {
        public List<Item> Items { get; set; }

        public void OnGet()
        {
            Items = new List<Item>
            {
                new Item { Sno = 1, Name = "Apple", Price = 20 },
                new Item { Sno = 2, Name = "Mango", Price = 15 },
                new Item { Sno = 3, Name = "Banana", Price = 3 }
            };
        }

        public class Item
        {
            public int Sno { get; set; }
            public string Name { get; set; }
            public int Price { get; set; }
        }
    }
}

üìÑ index.cshtml

@page
@model InvoiceApp.Pages.IndexModel
@{
    ViewData["Title"] = "Invoice Page";
}

<div class="text-center">
    <h1 class="display-4">Welcome</h1>
    <p>Learn about 
        <a href="https://docs.microsoft.com/aspnet/core">
            building Web apps with ASP.NET Core
        </a>
    </p>

    <h2>Invoice</h2>
    <table border="1" width="300">
        <caption>Invoice</caption>
        <tr>
            <th>Sno</th>
            <th>Item</th>
            <th>Price (Rs.)</th>
        </tr>
        @foreach (var item in Model.Items)
        {
            <tr>
                <td>@item.Sno</td>
                <td>@item.Name</td>
                <td>Rs. @item.Price</td>
            </tr>
        }
    </table>
</div>
Run the app:
  1. Navigate into the project folder: cd InvoiceApp
  2. Start the server: dotnet run
  3. Your site will be hosted at: http://localhost:5048 (or similar)
